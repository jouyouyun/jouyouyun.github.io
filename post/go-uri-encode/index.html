package utils

import (
        "net/url"
)

/**
 * Input: filepath
 * Output: encode string
 */
func encodeURI (filepath string) string {
        if len(filepath) < 1 {
                return ""
        }

        u := url.URL{}
        u.Path = filepath
        return u.String()
}

/**
 * Input: URI string
 * Output: filepath
 */
func decodeURI (uri string) string {
        if len(uri) < 1 {
                return ""
        }

        u, err := url.Parse(uri)
        if err != nil {
                return ""
        }

        return u.Path
}
