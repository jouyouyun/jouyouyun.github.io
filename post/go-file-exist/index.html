Go 笔记: 判断文件是否存在

jouyouyun

六月 20, 2014

使用 GoLang 判断文件是否存在。

package utils

import (
        "os"
)

func isFileExist(filename string) bool {
        if len(filename) < 1 {
                return false
        }

        _, err := os.Stat(filename)

        return err == nil || os.IsExist(err)
}
